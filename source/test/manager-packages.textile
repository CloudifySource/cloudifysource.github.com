function track_download(downloadHref) {
		var downloadme=downloadHref.split("cloudify-")[1];

		$created: new Date(),
		});
		mixpanel.track('Resource', {'Zip Name':zipName, 'Source':'Download Page'});
		mixpanel.people.increment({
			"Number Of Downloads": 1
		}); 
	}
